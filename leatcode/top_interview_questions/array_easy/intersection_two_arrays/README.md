class Solution:
    def intersect(self, nums1: list[int], nums2: list[int]) -> list[int]:
        # λ‘ λ°°μ—΄ μ •λ ¬
        nums1.sort()
        nums2.sort()
        
        i, j = 0, 0
        res = []
        
        # ν¬ ν¬μΈν„° μ΄λ™
        while i < len(nums1) and j < len(nums2):
            if nums1[i] < nums2[j]:
                i += 1
            elif nums1[i] > nums2[j]:

                i += 1
                j += 1
                
        return res
